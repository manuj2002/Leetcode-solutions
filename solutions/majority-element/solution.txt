class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int item=INT_MIN;
        int cnt=0;
        for(auto e:nums){
            if(item!=e && cnt==0){
                cnt++;
                item=e;
            }else if(item!=e){
                cnt--;
            }else{
                cnt++;
            }
        }

        return item;
    }
};
