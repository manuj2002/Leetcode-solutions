class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
       map<int,int>x;
       for(int i=0;i<nums.size();i++){
           x[nums[i]]=i;
       }

       vector<int>ans;
        for(int i=0;i<nums.size();i++){
           int y=target-nums[i];
           if(x.contains(y) && x[y]!=i){
            
             ans.push_back(x[y]);
             ans.push_back(i);
             break;
           }
       }
       return ans;
    }
};
