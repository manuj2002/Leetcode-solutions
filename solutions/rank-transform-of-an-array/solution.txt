class Solution {
public:
    vector<int> arrayRankTransform(vector<int>& arr) {
        int n=arr.size();
        vector<pair<int,int>>ar;
        for(int i=0;i<n;i++){
            ar.push_back({arr[i],i});
        }
        sort(ar.begin(),ar.end());
        int j=0;
        vector<int>ans(n,0);
        for(int i=0;i<n;i++){
            if(!(i-1>=0 && ar[i-1].first==ar[i].first)){
                j++;
            }
            ans[ar[i].second]=j;
        }

        return ans;
    }
};
