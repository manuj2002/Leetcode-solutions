class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int current_sum=nums[0], ans=nums[0];
        for(int i=1;i<nums.size();i++){
              if(nums[i]>current_sum){
                current_sum=nums[i];
              }else{
                 current_sum+=nums[i];
              }
              ans=max(ans,current_sum);
        }

        return ans;

    }
};
